#include <stdio.h>
#include <stdlib.h>
#define MALLOC_ERROR printf("Memory allocation error\n")

int main(void) {
	int* arr;
	int* ans;
	int* st1;
	int* st2;
	char* str;
	char c;
	int n, x, i, j;
	
	scanf("%d", &n);
	
	arr = (int*)malloc(n * sizeof(int));
	str = (char*)malloc(10 * n * sizeof(char));
	if (arr == NULL || str == NULL) {
		MALLOC_ERROR;
		return 1;
	}
	
	getchar();
	scanf("%[0-9 ]", str);
	
	x = 0;
	i = 0;
	j = 0;
	
	while (1) {
		c = str[j++];
		if (c >= '0' && c <= '9') {
			x = (x << 3) + (x << 1) + c - '0';
		}
		else {
			arr[i++] = x;
			x = 0;
			if (!c) {
				break;
			}
		}
	}
	
	free(str);
	
	ans = (int*)malloc(n * sizeof(int));
	st1 = (int*)malloc(n * sizeof(int));
	st2 = (int*)malloc(n * sizeof(int));
	if (ans == NULL || st1 == NULL || st2 == NULL) {
		MALLOC_ERROR;
		return 2;
	}
	
	for (i = 0; i < n; i++) {
		ans[i] = 0;
	}
	
	j = -1;
	for (i = 0; i < n; i++) {
		while (j >= 0) {
			if (st1[j] > arr[i]) {
				ans[i] = st2[j];
				break;
			}
			j--;
		}
		
		j++;
		st1[j] = arr[i];
		st2[j] = i + 1;
	}
	
	for (i = 0; i < n; i++) {
		printf("%d ", ans[i]);
	}
	printf("\n");
	
	free(arr);
	free(ans);
	free(st1);
	free(st2);
	
	return 0;
}
